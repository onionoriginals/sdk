    Interface AssetMigratedEvent

    Emitted when an asset migrates between layers

    Note: The details field is populated differently based on the target layer:

    • For did:webvh migrations: details is undefined (web publishing has no transaction details)
    • For did:btco migrations: details includes Bitcoin transaction information (transactionId, inscriptionId, satoshi, commitTxId, revealTxId, feeRate)
    interface AssetMigratedEvent {
        asset: { fromLayer: LayerType; id: string; toLayer: LayerType };
        details?: {
            commitTxId?: string;
            feeRate?: number;
            inscriptionId?: string;
            revealTxId?: string;
            satoshi?: string;
            transactionId?: string;
        };
        timestamp: string;
        type: "asset:migrated";
    }
